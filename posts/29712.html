安卓动态添加Fragment如何获取Fragment中的控件

Android Android Studio Fragment
发布日期:   2022-12-08
更新日期:   2022-12-08
阅读次数:  

动态添加多个Fragment 获取Fragment控件并监听

动态添加fragment 获取fragment的解决思路
（
将所有fragment添加到activity中
必须在onCreate中初始化所有fragment
再在onStart中getView (否则空指针警告)
）

package cn.itach.telegram;

import androidx.appcompat.app.AppCompatActivity;
import androidx.fragment.app.Fragment;
import androidx.fragment.app.FragmentManager;
import androidx.fragment.app.FragmentTransaction;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.LinearLayout;

import cn.itach.telegram.LoginUi.LoginFragment;
import cn.itach.telegram.LoginUi.LoginedFragment;

public class MainActivity extends AppCompatActivity implements View.OnClickListener {
    private Button login_btn;
    private Button back_login_btn;
    private Fragment loginFragment;
    private Fragment loginedFragment;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        // 在onCreate初始化Fragment
        loginFragment = new LoginFragment();
        loginedFragment = new LoginedFragment();
        // 将所有fragment添加到MainActivity中
        addAllLoginFragment();
        showLogin();

    }

    @Override
    protected void onStart() {
        super.onStart();
        initFragmentListener();
    }

    private void initFragmentListener() {
        // 在onStart中获取到所有fragment的控件
        login_btn = loginFragment.getView().findViewById(R.id.login_btn);
        back_login_btn = loginedFragment.getView().findViewById(R.id.back_login);
        // 监听fragmenet中控件的点击事件
        login_btn.setOnClickListener(this);
        back_login_btn.setOnClickListener(this);

    }

    // 将所有fragment添加到MainActivity中
    public void addAllLoginFragment() {
        FragmentManager fragmentManager = getSupportFragmentManager();
        FragmentTransaction transaction = fragmentManager.beginTransaction();
        transaction.add(R.id.login_body,loginFragment).add(R.id.login_body,loginedFragment);
        transaction.commit();
    }

    // 控制两个fragment的隐藏与显示
    private void showLogin() {
        FragmentManager fragmentManager = getSupportFragmentManager();
        FragmentTransaction transaction = fragmentManager.beginTransaction();
        transaction.hide(loginedFragment);
        transaction.show(loginFragment);
        transaction.commit();
    }
    // 控制两个fragment的隐藏与显示
    private void showLogined() {
        FragmentManager fragmentManager = getSupportFragmentManager();
        FragmentTransaction transaction = fragmentManager.beginTransaction();
        transaction.hide(loginFragment);
        transaction.show(loginedFragment);
        transaction.commit();
    }

    // 监听点击事件
    @Override
    public void onClick(View view) {
        switch (view.getId()) {
            case R.id.login_btn:
                showLogined();
                break;
            case R.id.back_login:
                showLogin();
                break;
        }
    }
}
